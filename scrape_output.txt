import React, { useState, useRef, useEffect } from 'react';
import { View, Text, StyleSheet, FlatList, KeyboardAvoidingView, Platform, TouchableOpacity } from 'react-native';
import { TextInput, Button, Card, ActivityIndicator } from 'react-native-paper';
import { geminiAdvisory, getRuleBasedAdvisory } from '../services/aiService';
import { getCurrentUser } from '../services/firebase';
export default function AdvisoryScreen() {
  const [messages, setMessages] = useState([
    { id: '1', text: 'আসসালামু আলাইকুম! আমি আপনার কৃষি সহায়ক। আপনাকে কীভাবে সাহায্য করতে পারি?', sender: 'bot', timestamp: new Date() }
  ]);
  const [inputText, setInputText] = useState('');
  const [loading, setLoading] = useState(false);
  const [cropType, setCropType] = useState('ধান');
  const flatListRef = useRef();
  const quickQuestions = [
    'ধানের পোকার সমাধান',
    'আলুর রোগ চিকিৎসা',
    'সার প্রয়োগের নিয়ম',
    'সেচ ব্যবস্থাপনা'
  ];
  const sendMessage = async (text = inputText) => {
    if (!text.trim()) return;
    const userMessage = {
      id: Date.now().toString(),
      text: text,
      sender: 'user',
      timestamp: new Date()
    };
    setMessages(prev => [...prev, userMessage]);
    setInputText('');
    setLoading(true);
    try {
      const response = await geminiAdvisory(text, cropType);
      
      const botMessage = {
        id: (Date.now() + 1).toString(),
        text: response,
        sender: 'bot',
        timestamp: new Date()
      };
      setMessages(prev => [...prev, botMessage]);
    } catch (error) {
      const errorMessage = {
        id: (Date.now() + 1).toString(),
        text: 'দুঃখিত, কিছু একটা সমস্যা হয়েছে। অনুগ্রহ করে আবার চেষ্টা করুন।',
        sender: 'bot',
        timestamp: new Date()
      };
      setMessages(prev => [...prev, errorMessage]);
    } finally {
      setLoading(false);
    }
  };
  const renderMessage = ({ item }) => (
    <View style={[
      styles.messageContainer,
      item.sender === 'user' ? styles.userMessage : styles.botMessage
    ]}>
      <Text style={[
        styles.messageText,
        item.sender === 'user' ? styles.userMessageText : styles.botMessageText
      ]}>
        {item.text}
      </Text>
      <Text style={styles.timestamp}>
        {item.timestamp.toLocaleTimeString('bn-BD', { hour: '2-digit', minute: '2-digit' })}
      </Text>
    </View>
  );
  return (
    <KeyboardAvoidingView 
      behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
      style={styles.container}
    >
      {/* Header */}
      <View style={styles.header}>
        <Text style={styles.headerTitle}>💬 কৃষি চ্যাটবট</Text>
        <View style={styles.cropSelector}>
          <Text style={styles.cropLabel}>ফসল:</Text>
          <TouchableOpacity 
            style={styles.cropButton}
            onPress={() => {
              const crops = ['ধান', 'গম', 'আলু', 'টমেটো', 'মরিচ', 'লাউ'];
              const currentIndex = crops.indexOf(cropType);
              setCropType(crops[(currentIndex + 1) % crops.length]);
            }}
          >
            <Text style={styles.cropText}>{cropType}</Text>
          </TouchableOpacity>
        </View>
      </View>
      {/* Quick Questions */}
      <View style={styles.quickQuestions}>
        <Text style={styles.quickLabel}>দ্রুত প্রশ্ন:</Text>
        <ScrollView horizontal showsHorizontalScrollIndicator={false}>
          {quickQuestions.map((q, index) => (
            <Button 
              key={index}
              mode="outlined"
              onPress={() => sendMessage(q)}
              style={styles.quickButton}
              contentStyle={styles.quickButtonContent}
            >
              {q}
            </Button>
          ))}
        </ScrollView>
      </View>
      {/* Messages */}
      <FlatList
        ref={flatListRef}
        data={messages}
        renderItem={renderMessage}
        keyExtractor={item => item.id}
        style={styles.messagesList}
        contentContainerStyle={styles.messagesContent}
        onContentSizeChange={() => flatListRef.current?.scrollToEnd({ animated: true })}
      />
      {/* Loading Indicator */}
      {loading && (
        <View style={styles.loadingContainer}>
          <ActivityIndicator size="small" color=" #2E7D32" />
          <Text style={styles.loadingText}>চিন্তা করছি...</Text>
        </View>
      )}
      {/* Input Area */}
      <View style={styles.inputContainer}>
        <TextInput
          value={inputText}
          onChangeText={setInputText}
          placeholder="আপনার প্রশ্ন লিখুন..."
          mode="outlined"
          style={styles.input}
          multiline
          maxLength={500}
        />
        <Button 
          mode="contained" 
          onPress={() => sendMessage()}
          disabled={!inputText.trim() || loading}
          style={styles.sendButton}
        >
          পাঠান
        </Button>
      </View>
    </KeyboardAvoidingView>
  );
}

import React, { useState } from 'react';
import { View, Text, StyleSheet, Image, ScrollView, TouchableOpacity, Alert } from 'react-native';
import { Button, Card, ActivityIndicator } from 'react-native-paper';
import * as ImagePicker from 'expo-image-picker';
import { detectPestDisease } from '../services/aiService';
export default function PestDetectionScreen() {
  const [image, setImage] = useState(null);
  const [result, setResult] = useState(null);
  const [loading, setLoading] = useState(false);
  const pickImage = async () => {
    const permissionResult = await ImagePicker.requestMediaLibraryPermissionsAsync();
    
    if (permissionResult.granted === false) {
      Alert.alert('অনুমতি প্রয়োজন', 'ছবি তোলার জন্য ক্যামেরা অনুমতি প্রয়োজন');
      return;
    }
    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true,
      aspect: [4, 3],
      quality: 0.8,
    });
    if (!result.canceled) {
      setImage(result.assets[0].uri);
      setResult(null);
    }
  };
  const takePhoto = async () => {
    const permissionResult = await ImagePicker.requestCameraPermissionsAsync();
    
    if (permissionResult.granted === false) {
      Alert.alert('অনুমতি প্রয়োজন', 'ছবি তোলার জন্য ক্যামেরা অনুমতি প্রয়োজন');
      return;
    }
    const result = await ImagePicker.launchCameraAsync({
      allowsEditing: true,
      aspect: [4, 3],
      quality: 0.8,
    });
    if (!result.canceled) {
      setImage(result.assets[0].uri);
      setResult(null);
    }
  };
  const analyzeImage = async () => {
    if (!image) {
      Alert.alert('ছবি নির্বাচন করুন', 'অনুগ্রহ করে একটি ছবি নির্বাচন করুন');
      return;
    }
    setLoading(true);
    try {
      const detectionResult = await detectPestDisease(image);
      setResult(detectionResult);
    } catch (error) {
      Alert.alert('ত্রুটি', 'বিশ্লেষণ করতে সমস্যা হয়েছে। আবার চেষ্টা করুন।');
    } finally {
      setLoading(false);
    }
  };
  const resetAnalysis = () => {
    setImage(null);
    setResult(null);
  };
  return (
    <ScrollView style={styles.container}>
      {/* Header */}
      <View style={styles.header}>
        <Text style={styles.title}>📸 রোগ ও পোকা শনাক্তকরণ</Text>
        <Text style={styles.subtitle}>আপনার ফসলের ছবি তুলুন এবং AI দিয়ে বিশ্লেষণ করুন</Text>
      </View>
      {/* Instructions */}
      <Card style={styles.card}>
        <Card.Content>
          <Text style={styles.instructionTitle}>কীভাবে ব্যবহার করবেন:</Text>
          <Text style={styles.instruction}>1. ফসলের আক্রান্ত অংশের পরিষ্কার ছবি তুলুন</Text>
          <Text style={styles.instruction}>2. ভালো আলোতে ছবি তুলুন</Text>
          <Text style={styles.instruction}>3. কাছ থেকে ছবি তুলুন যাতে রোগ/পোকা স্পষ্ট দেখা যায়</Text>
        </Card.Content>
      </Card>
      {/* Action Buttons */}
      <View style={styles.actionButtons}>
        <Button 
          mode="contained" 
          onPress={takePhoto}
          icon="camera"
          style={styles.actionButton}
          disabled={loading}
        >
          ক্যামেরা দিয়ে ছবি তুলুন
        </Button>
        
        <Button 
          mode="outlined" 
          onPress={pickImage}
          icon="image"
          style={styles.actionButton}
          disabled={loading}
        >
          গ্যালারি থেকে নির্বাচন
        </Button>
      </View>
      {/* Image Preview */}
      {image && (
        <Card style={styles.card}>
          <Card.Content>
            <Text style={styles.sectionTitle}>নির্বাচিত ছবি:</Text>
            <Image source={{ uri: image }} style={styles.image} />
            
            {!result && (
              <Button 
                mode="contained" 
                onPress={analyzeImage}
                loading={loading}
                disabled={loading}
                style={styles.analyzeButton}
              >
                {loading ? 'বিশ্লেষণ হচ্ছে...' : 'বিশ্লেষণ করুন'}
              </Button>
            )}
          </Card.Content>
        </Card>
      )}
      {/* Results */}
      {result && (
        <Card style={[styles.card, styles.resultCard]}>
          <Card.Title 
            title={result.detected ? '✅ বিশ্লেষণ সম্পন্ন' : '⚠️ সীমিত ফলাফল'}
            titleStyle={{ color: result.detected ? ' #2E7D32' : ' #FF9800' }}
          />
          <Card.Content>
            <Text style={styles.resultText}>{result.diagnosis}</Text>
            
            {result.confidence > 0 && (
              <Text style={styles.confidence}>
                আস্থার মাত্রা: {(result.confidence * 100).toFixed(0)}%
              </Text>
            )}
            <View style={styles.resultActions}>
              <Button 
                mode="outlined" 
                onPress={resetAnalysis}
                style={styles.resultButton}
              >
                নতুন ছবি
              </Button>
              <Button 
                mode="contained" 
                onPress={() => Alert.alert('সহায়তা', 'স্থানীয় কৃষি অফিসারের সাথে যোগাযোগ করুন')}
                style={styles.resultButton}
              >
                বিশেষজ্ঞের পরামর্শ
              </Button>
            </View>
          </Card.Content>
        </Card>
      )}
      {/* Sample Images */}
      <Card style={styles.card}>
        <Card.Title title="📚 নমুনা ছবি" />
        <Card.Content>
          <Text style={styles.sampleText}>
            সঠিক ছবি: আক্রান্ত পাতা/ফলের কাছ থেকে পরিষ্কার ছবি
          </Text>
          <Text style={styles.sampleText}>
            ভুল ছবি: দূর থেকে ঝাপসা ছবি বা খুব বেশি আলো/অন্ধকার
          </Text>
        </Card.Content>
      </Card>
    </ScrollView>
  );
}
const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: ' #f5f5f5' },

import React, { useEffect, useState } from 'react';
import { StatusBar } from 'expo-status-bar';
import { PaperProvider } from 'react-native-paper';
import { onAuthStateChanged } from 'firebase/auth';
import { auth } from './src/services/firebase';
